#include<iostream>
using namespace std;
int main()
{
    char str[100];
    int len = 0;
    bool flag = true;
    int i = 0;
    str[i] = getchar();
    while (str[i] != '\n')
    {
        i++;
        str[i] = getchar();
    }
    len = i - 1;
    for (int j = 0; j < (len / 2); j++)
    {
        if (str[j] != str[len - j])
        {
            flag = false;
            break;
        }
    }
    if (false == flag)
    {
        printf("False");
    }
    else
    {
        printf("True");
    }
    return 0;
}
