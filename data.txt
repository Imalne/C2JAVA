#include <iostream>
#include <cstring>

int main() {
    char text[] = "anpanman";
    char pattern[] = "an";

    int answer[100];
    int a = 0;
    int m = 8;
    int n = 2;
    int i = 0;
    int j = -1;
    int next[100];
    next[0] = -1;
    while (i < n) {
        while (j > -1 && pattern[i] != pattern[j]) {
            j = next[j];
        }
        i += 1;
        j += 1;
        if ((i != n) && (j != m) && (pattern[i] == pattern[j])){
            next[i] = next[j];
        }
        else if ((i == n) && (j == m)){
            next[i] = next[j];
        }
        else{
            next[i] = j;
        }
    }

    i = 0;
    j = 0;
    while (j < m) {
        while (i > -1 && text[j] != pattern[i]){
            i = next[i];
        }
        i += 1;
        j += 1;
        if (i >= n) {
            answer[a] = j - i;
            a += 1;
            i = next[i];
        }
    }
    printf("TEXT: '%s', PATTERN: '%s' ANSWER: %c", text, pattern, 10);
    if(a > 0){
        for(i = 0; i < a; i += 1){
            printf("index %d ", answer[i]);
        }
        printf(" %c", 10);
    }
    else{
        printf("False %c", 10);
    }
    return 0;
}
